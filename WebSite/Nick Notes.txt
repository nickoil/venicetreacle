2. CSS screen size rules

.myTableBg{
    text-align: center;
    background: url(yellow80px.png) repeat;
    border-collapse: collapse;
    font-size: large;
}

.myTableBg td, .myTableBg th {
    border:0;
    padding:5px;
}

/* if the screen is < 480px wide e.g. portrait mobile use these rules as well or instead */

@media (max-width: 480px) {
    .myTableBg {
        text-align: left;
        background: none;
        padding: 20px;
        font-size: xx-large;
        margin-left: 20px;
    }
    
    .myTableBg td, .myTableBg th {
        padding:20px;
        margin-left: 20px;
    }
}
